Family Tree using prolog

% Define relationships
parent(john, mary).
parent(mary, susan).
parent(susan, mike).
parent(mike, alex).

% Define gender
male(john).
female(mary).
female(susan).
male(mike).
male(alex).

% Define rules
mother(Mother, Child) :- parent(Mother, Child), female(Mother).
father(Father, Child) :- parent(Father, Child), male(Father).

% Siblings are people who share at least one parent
sibling(Sibling1, Sibling2) :- parent(Parent, Sibling1), parent(Parent, Sibling2), Sibling1 \= Sibling2.

% Grandparents are parents of a parent
grandparent(Grandparent, Grandchild) :- parent(Grandparent, Parent), parent(Parent, Grandchild).

% Query examples
% ?- mother(Mary, Child).
% ?- father(John, Child).
% ?- sibling(Mike, Alex).
% ?- grandparent(Grandparent, Alex).
