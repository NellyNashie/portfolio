Grading System using python programming

def get_grade(score):
if score >= 90:
return 'A'
elif score >= 80:
return 'B'
elif score >= 70:
return 'C'
elif score >= 60:
return 'D'
else:
return 'F'

def main():
print("Grading System")
name = input("Enter student name: ")
score = float(input("Enter student score (0-100): "))

if 0 <= score <= 100:
grade = get_grade(score)
print(f"{name} received a grade of {grade}")
else:
print("Invalid score. Please enter a score between 0 and 100.")

if __name__ == "__main__":
main()
